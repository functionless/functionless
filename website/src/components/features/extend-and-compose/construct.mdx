export class Cart extends Construct {
  constructor(scope: Construct = getScope(), id: string = getId()) {
    super(scope, id);

    this.cartTable = new Table(this, "cartTable");

    this.getCart = new Function(this, "getCart", async (cartId) => {
      return this.cartTable.getItem(cartId);
    }
  }
}
